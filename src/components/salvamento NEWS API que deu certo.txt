import React, { useEffect, useState } from "react";
import axios from "axios";
import {
  ProductBox,
  ImageBox,
  Image,
  Title,
  Description,
  InitialPrice,
  FinalPrice,
  Installments
} from "./LandingPageStyle";

const LandingPage = () => {
  const url =
    // "https://frontend-intern-challenge-api.iurykrieger.now.sh/products?page=1";
    "https://newsapi.org/v2/everything?q=world&language=pt&apiKey=c452469e67b54b90918dc69924b83d36";

  const [products, setProducts] = useState([]);
  let productsForMap = products;

  useEffect(() => {
    axios
      .get(url)
      .then((response) => {
        console.log("Response");
        console.log(response);

        setProducts(response.data.articles);
      })
      .catch((error) => {
        console.error("catch erro " + error);
      });
  }, []);
  console.log("Product");
  console.log(products);
  return (
    <>
      {products.map((product, index) => {
        return (
          <ProductBox>
            <ImageBox>
              <Image
                src={product.urlToImage}
                onClick={() => {
                  window.open(product.url);
                }}
              ></Image>
            </ImageBox>

            <Title>{product.title}</Title>
            <Description></Description>
            <InitialPrice></InitialPrice>
            <FinalPrice></FinalPrice>
            <Installments></Installments>
          </ProductBox>
        );
      })}
    </>

};

export default LandingPage;
